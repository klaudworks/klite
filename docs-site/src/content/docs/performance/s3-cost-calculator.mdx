### Pricing reference

| Operation | Rate |
|-----------|------|
| PUT / COPY / POST / LIST | $5.00 per million |
| GET / SELECT | $0.40 per million |
| DELETE | Free |
| Storage (first 50 TB) | $0.023 per GB-month |

### What drives costs

**Storage** is the dominant cost in almost every scenario (90%+ of total). API costs only become significant at very high message rates with multiple S3-tier consumers.

After the listing cache append optimization, **PUTs are the largest API cost**. Each flush cycle creates one S3 object per partition (when data exceeds the target object size or flush interval). Increasing the target object size (`--s3-target-object-size`) reduces PUT count proportionally.

**GETs** scale with `objects * consumers` — each S3-tier consumer reads each new object once (range GET + footer read).

**LISTs** are now negligible. The listing cache is maintained in-memory and only refreshed on cold start or after compaction.

<script is:inline>{`
  const M = 2592000; // seconds per month
  const PUT_RATE = 0.005 / 1000;
  const GET_RATE = 0.0004 / 1000;
  const LIST_RATE = 0.005 / 1000;
  const STORAGE_RATE = 0.023; // per GB-month

  function fmt(n) {
    if (n >= 1e9) return (n / 1e9).toFixed(1) + 'B';
    if (n >= 1e6) return (n / 1e6).toFixed(1) + 'M';
    if (n >= 1e3) return (n / 1e3).toFixed(1) + 'K';
    return n.toFixed(0);
  }
  function usd(n) { return '$' + n.toFixed(2); }

  function calculate() {
    var rate = Number(document.getElementById('msgRate').value) || 0;
    var size = Number(document.getElementById('msgSize').value) || 0;
    var P = Number(document.getElementById('partitions').value) || 1;
    var C = Number(document.getElementById('consumers').value) || 0;
    var D = Number(document.getElementById('retention').value) || 1;
    var targetMiB = Number(document.getElementById('objectSize').value) || 64;
    var T = Number(document.getElementById('flushInterval').value) || 60;

    var comp = Math.max(1, Number(document.getElementById('compression').value) || 1);

    var targetBytes = targetMiB * 1024 * 1024;
    var bytesPerSecUncompressed = rate * size;
    var bytesPerSec = bytesPerSecUncompressed / comp;
    var bytesPerPartitionPerInterval = (bytesPerSec / P) * T;
    var objectsPerInterval = Math.max(1, bytesPerPartitionPerInterval / targetBytes);
    var flushPUTs = objectsPerInterval * P * (M / T);

    // PUTs: flush + metadata.log + lease renewal
    var metaPUTs = M / T;
    var leasePUTs = M / 5;
    var totalPUTs = flushPUTs + metaPUTs + leasePUTs;

    // GETs: consumer range + footer + lease standby
    var consumerGETs = flushPUTs * C * 2; // range + footer
    var leaseGETs = M / 2;
    var totalGETs = consumerGETs + leaseGETs;

    // LISTs: negligible after listing cache append optimization
    // retention (per partition per hour) + compaction + GC
    var retentionLISTs = P * (M / 3600);
    var gcLISTs = M / 30;
    var totalLISTs = retentionLISTs + gcLISTs;

    // Storage
    var storageBytes = bytesPerSec * D * 86400;
    var storageGB = storageBytes / (1024 * 1024 * 1024);

    var putCost = totalPUTs * PUT_RATE;
    var getCost = totalGETs * GET_RATE;
    var listCost = totalLISTs * LIST_RATE;
    var storageCostVal = storageGB * STORAGE_RATE;
    var total = putCost + getCost + listCost + storageCostVal;

    document.getElementById('putCount').textContent = fmt(totalPUTs);
    document.getElementById('putCost').textContent = usd(putCost);
    document.getElementById('getCount').textContent = fmt(totalGETs);
    document.getElementById('getCost').textContent = usd(getCost);
    document.getElementById('listCount').textContent = fmt(totalLISTs);
    document.getElementById('listCost').textContent = usd(listCost);
    var storageTB = storageGB / 1024;
    document.getElementById('storageSize').textContent = storageGB >= 1024
      ? storageTB.toFixed(1) + ' TB'
      : fmt(storageGB) + ' GB';
    document.getElementById('storageCost').textContent = usd(storageCostVal);
    document.getElementById('totalCost').textContent = usd(total);
    document.getElementById('dataRate').textContent = (bytesPerSecUncompressed / (1024 * 1024)).toFixed(1) + ' MiB/s';
    document.getElementById('diskRate').textContent = (bytesPerSec / (1024 * 1024)).toFixed(1) + ' MiB/s';
    document.getElementById('objectCount').textContent = fmt(flushPUTs);
  }

  document.querySelectorAll('#calc input').forEach(function(el) {
    el.addEventListener('input', calculate);
  });
  calculate();
`}</script>
