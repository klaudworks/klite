### What drives costs

**Storage dominates** — typically 90%+ of total cost. API costs only matter at very high throughput.

The main API cost driver is **object store reads**: consumers fetching data that's been evicted from the on-disk WAL because `KLITE_WAL_MAX_DISK_SIZE` was exceeded and must be read back from object storage.

**Data transfer**: if klite runs in the same region as the bucket, there are no transfer costs. Running outside the provider's network incurs egress fees (<span id="egressNote">~$0.09/GB</span>) when consumers fetch WAL-evicted data. Writes are always free (ingress is $0).

<script is:inline>{`
  var M = 2592000;

  var PROVIDERS = {
    aws: {
      put: 0.005 / 1000, get: 0.0004 / 1000, list: 0.005 / 1000, storage: 0.023,
      note: 'US East (N. Virginia), S3 Standard.',
      egress: '~$0.09/GB',
      putLabel: 'PUT', getLabel: 'GET', listLabel: 'LIST',
      pricingRows: [
        ['PUT / COPY / POST / LIST', '$5.00 per million'],
        ['GET / SELECT', '$0.40 per million'],
        ['Delete', 'Free'],
        ['Storage (first 50 TB)', '$0.023 per GB-month']
      ]
    },
    gcs: {
      put: 0.005 / 1000, get: 0.0004 / 1000, list: 0.005 / 1000, storage: 0.020,
      note: 'US multi-region / us-central1, Standard class. Same pricing via S3-compatible XML API.',
      egress: '~$0.12/GB',
      putLabel: 'Class A (write)', getLabel: 'Class B (read)', listLabel: 'Class A (list)',
      pricingRows: [
        ['Class A (write / list)', '$5.00 per million'],
        ['Class B (read)', '$0.40 per million'],
        ['Delete', 'Free'],
        ['Storage (first 50 TB)', '$0.020 per GB-month']
      ]
    },
    azure: {
      put: 0.0081 / 1000, get: 0.0005 / 1000, list: 0.0081 / 1000, storage: 0.0184,
      note: 'East US, Hot tier, LRS redundancy.',
      egress: '~$0.087/GB',
      putLabel: 'Write (Put)', getLabel: 'Read (Get)', listLabel: 'Write (List)',
      pricingRows: [
        ['Write (Put, List)', '$8.10 per million'],
        ['Read (Get)', '$0.50 per million'],
        ['Delete', 'Free'],
        ['Storage (first 50 TB)', '$0.0184 per GB-month']
      ]
    }
  };

  var currentProvider = 'aws';

  function fmt(n) {
    if (n >= 1e9) return (n / 1e9).toFixed(1) + 'B';
    if (n >= 1e6) return (n / 1e6).toFixed(1) + 'M';
    if (n >= 1e3) return (n / 1e3).toFixed(1) + 'K';
    return n.toFixed(0);
  }
  function usd(n) { return '$' + n.toFixed(2); }

  function renderPricingTable() {
    var p = PROVIDERS[currentProvider];
    var html = '<table><thead><tr><th>Operation</th><th>Rate</th></tr></thead><tbody>';
    p.pricingRows.forEach(function(row) {
      html += '<tr><td>' + row[0] + '</td><td>' + row[1] + '</td></tr>';
    });
    html += '</tbody></table>';
    document.getElementById('pricingTable').innerHTML = html;
  }

  function updateLabels() {
    var p = PROVIDERS[currentProvider];
    document.getElementById('providerNote').textContent = p.note;
    document.getElementById('egressNote').textContent = p.egress;
    document.getElementById('putLabel').textContent = p.putLabel;
    document.getElementById('getLabel').textContent = p.getLabel;
    document.getElementById('listLabel').textContent = p.listLabel;
    renderPricingTable();
  }

  function calculate() {
    var p = PROVIDERS[currentProvider];
    var rate = Number(document.getElementById('msgRate').value) || 0;
    var size = Number(document.getElementById('msgSize').value) || 0;
    var P = Number(document.getElementById('partitions').value) || 1;
    var C = Number(document.getElementById('consumers').value) || 0;
    var D = Number(document.getElementById('retention').value) || 1;
    var targetMiB = Number(document.getElementById('objectSize').value) || 64;
    var T = Number(document.getElementById('flushInterval').value) || 60;
    var comp = Math.max(1, Number(document.getElementById('compression').value) || 1);

    var targetBytes = targetMiB * 1024 * 1024;
    var bytesPerSecUncompressed = rate * size;
    var bytesPerSec = bytesPerSecUncompressed / comp;
    var bytesPerPartitionPerInterval = (bytesPerSec / P) * T;
    var objectsPerInterval = Math.max(1, bytesPerPartitionPerInterval / targetBytes);
    var flushPUTs = objectsPerInterval * P * (M / T);

    var metaPUTs = M / T;
    var leasePUTs = M / 5;
    var totalPUTs = flushPUTs + metaPUTs + leasePUTs;

    var consumerGETs = flushPUTs * C * 2;
    var leaseGETs = M / 2;
    var totalGETs = consumerGETs + leaseGETs;

    var retentionLISTs = P * (M / 3600);
    var gcLISTs = M / 30;
    var totalLISTs = retentionLISTs + gcLISTs;

    var storageBytes = bytesPerSec * D * 86400;
    var storageGB = storageBytes / (1024 * 1024 * 1024);

    var putCost = totalPUTs * p.put;
    var getCost = totalGETs * p.get;
    var listCost = totalLISTs * p.list;
    var storageCostVal = storageGB * p.storage;
    var total = putCost + getCost + listCost + storageCostVal;

    document.getElementById('putCount').textContent = fmt(totalPUTs);
    document.getElementById('putCost').textContent = usd(putCost);
    document.getElementById('getCount').textContent = fmt(totalGETs);
    document.getElementById('getCost').textContent = usd(getCost);
    document.getElementById('listCount').textContent = fmt(totalLISTs);
    document.getElementById('listCost').textContent = usd(listCost);
    var storageTB = storageGB / 1024;
    document.getElementById('storageSize').textContent = storageGB >= 1024
      ? storageTB.toFixed(1) + ' TB'
      : fmt(storageGB) + ' GB';
    document.getElementById('storageCost').textContent = usd(storageCostVal);
    document.getElementById('totalCost').textContent = usd(total);
    document.getElementById('dataRate').textContent = (bytesPerSecUncompressed / (1024 * 1024)).toFixed(1) + ' MiB/s';
    document.getElementById('diskRate').textContent = (bytesPerSec / (1024 * 1024)).toFixed(1) + ' MiB/s';
    document.getElementById('objectCount').textContent = fmt(flushPUTs);
  }

  // Provider tab switching
  document.querySelectorAll('.provider-tab').forEach(function(tab) {
    tab.addEventListener('click', function() {
      document.querySelectorAll('.provider-tab').forEach(function(t) { t.classList.remove('active'); });
      tab.classList.add('active');
      currentProvider = tab.getAttribute('data-provider');
      updateLabels();
      calculate();
    });
  });

  document.querySelectorAll('#calc input').forEach(function(el) {
    el.addEventListener('input', calculate);
  });

  updateLabels();
  calculate();
`}</script>
